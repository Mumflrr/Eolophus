"""
Event Container

Container for event data including type, payload, and unique identifier.
Implements auto-ID generation via __post_init__ as per PlanSpec fix.
"""

from __future__ import annotations
from dataclasses import dataclass
from typing import Any, Optional
from uuid import uuid4


@dataclass
class Event:
    """
    Container for event data including type, payload, and unique identifier.

    Attributes:
        event_type: The type/name of the event
        payload: Arbitrary data to carry with the event
        id: Unique identifier (auto-generated if not provided)
    """
    event_type: str
    payload: Any
    id: Optional[str] = None

    def __post_init__(self) -> None:
        """
        Automatically generate a unique ID if not provided during initialization.

        This ensures every Event instance has a valid UUID without requiring
        manual intervention after construction. Fixes the issue where
        auto_generate_id() had to be called manually.
        """
        if self.id is None:
            self.id = str(uuid4())

    def __str__(self) -> str:
        """Human-readable string representation of the event."""
        return f"Event(id={self.id}, type={self.event_type})"

    def __repr__(self) -> str:
        """Developer-friendly representation of the event."""
        return f"Event(event_type='{self.event_type}', payload={self.payload!r}, id='{self.id}')"

"""
SimpleEventBus Core Orchestrator

Thread-safe, in-memory, synchronous event bus implementation
following the publish-subscribe pattern.
"""

from __future__ import annotations
from typing import Any, Callable
import threading


class SimpleEventBus:
    """
    Core orchestrator for event subscription, publishing, and listener lifecycle.

    This implementation provides:
    - Thread-safe operations via explicit locking
    - Exception isolation (listener errors don't affect others)
    - In-memory storage (no persistence)
    - Synchronous call execution

    Attributes:
        _listeners: Dictionary mapping event types to lists of callback functions
        _lock: Threading lock for thread-safe access to listeners
    """

    def __init__(self) -> None:
        """Initialize the event bus with internal state."""
        self._listeners: dict[str, list[Callable[..., Any]]] = {}
        self._lock: threading.Lock = threading.Lock()

    def subscribe(self, event_type: str, callback: Callable[..., Any]) -> None:
        """
        Register a listener callback for a specific event type.

        Args:
            event_type: The event type to listen for
            callback: Function to call when event is published

        Thread Safety:
            Protected by internal lock to prevent race conditions
            during concurrent subscription operations.
        """
        with self._lock:
            if event_type not in self._listeners:
                self._listeners[event_type] = []
            self._listeners[event_type].append(callback)

    def unsubscribe(self, event_type: str, callback: Callable[..., Any]) -> None:
        """
        Remove a specific listener from an event type.

        Args:
            event_type: The event type to unsubscribe from
            callback: The callback function to remove

        Notes:
            - If the callback was never added, this is handled gracefully
            - If the event_type has no listeners, this is handled gracefully
        """
        with self._lock:
            if event_type in self._listeners:
                try:
                    self._listeners[event_type].remove(callback)
                    # Clean up empty lists to prevent memory leaks
                    if not self._listeners[event_type]:
                        del self._listeners[event_type]
                except ValueError:
                    # Callback not in list, handle gracefully
                    pass

    def publish(self, event: Event) -> None:
        """
        Publish an event to all registered listeners.

        Args:
            event: The event to publish

        Thread Safety:
            - Lock is acquired to get a snapshot of listeners
            - Each listener is called independently
            - Exceptions in one listener don't affect others

        Notes:
            - Catches exceptions per listener to prevent cascade failures
            - Publishing to event type with no listeners is a no-op
        """
        # Get a snapshot of listeners while holding the lock
        with self._lock:
            listeners = self._listeners.get(event.event_type, []).copy()

        # Call each listener independently to prevent cascade failures
        for callback in listeners:
            try:
                callback(event)
            except Exception:
                # Log or handle the exception - listener errors should not block others
                # In production, you might want to log this for debugging
                pass

    def clear(self) -> None:
        """
        Remove all listeners from the bus.

        Useful for testing and resetting state.
        Thread-safe operation.
        """
        with self._lock:
            self._listeners.clear()
